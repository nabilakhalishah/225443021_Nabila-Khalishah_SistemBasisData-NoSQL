// agregasi via python
import os
from dotenv import load_dotenv
from pymongo import MongoClient

load_dotenv()
client = MongoClient(os.getenv("MONGO_URI"))
db = client[os.getenv("DB_NAME")]
collection = db["sensor"]

pipeline = [
    {"$group": {
        "_id": "$mesin",
        "rata_suhu": {"$avg": "$suhu"},
        "max_getaran": {"$max": "$getaran"},
        "count": {"$sum": 1}
    }},
    {"$sort": {"rata_suhu": -1}}
]

results = collection.aggregate(pipeline)
print("Rata-rata suhu per mesin:")
for doc in results:
    print(f"{doc['_id']} -> rata:{doc['rata_suhu']:.2f}°C, max getaran:{doc['max_getaran']}, jumlah data:{doc['count']}")

client.close()

import pandas as pd
import matplotlib.pyplot as plt
from dotenv import load_dotenv
from pymongo import MongoClient
import os

#Ambil Data
cursor = collection.find({}, {"_id": 0})
df = pd.DataFrame(list(cursor))

if df.empty:
    print("Data tidak ditemukan di database!")
else:

    df["timestamp"] = pd.to_datetime(df["timestamp"])
    
    # Filter mesin CNC-01
    df = df[df["mesin"] == "CNC-01"]
    
    # Urutkan berdasarkan waktu agar garis grafik tidak bolak-balik
    df = df.sort_values("timestamp")
    
    # Pastikan data suhu numerik
    df['suhu'] = pd.to_numeric(df['suhu'], errors='coerce')
    
    # Set index ke timestamp (untuk grafik dan resample)
    df.set_index("timestamp", inplace=True)

    # Agregasi (Sesuai Fase 4)
    resampled = df[['suhu']].resample('10min').mean()

    plt.figure(figsize=(10, 5))
    plt.plot(resampled.index, resampled["suhu"], marker='o', linestyle='-', color='b')
    
    plt.title("Tren Suhu CNC-01 (Rata-rata 10 Menit)")
    plt.xlabel("Waktu")
    plt.ylabel("Suhu (°C)")
    plt.grid(True)
    plt.xticks(rotation=45)
    plt.tight_layout() 
    
    print("Menampilkan grafik...")
    plt.show()

    # Ekspor ke CSV (PERBAIKAN DI SINI)
    resampled.to_csv("agregasi_cnc01.csv")
    print("Data hasil agregasi berhasil disimpan ke 'agregasi_cnc01.csv'")

import json, os
import paho.mqtt.client as mqtt
from pymongo import MongoClient
from datetime import datetime
from dotenv import load_dotenv

load_dotenv()
MONGO_URI = os.getenv("MONGO_URI")
DB_NAME = os.getenv("DB_NAME")

mongo_client = MongoClient(MONGO_URI)
db = mongo_client[DB_NAME]
collection = db["sensor"]

def on_connect(client, userdata, flags, rc):
    print("Terhubung ke MQTT broker")
    client.subscribe("pabrik/sensor/suhu")

def on_message(client, userdata, msg):
    try:
        payload = json.loads(msg.payload.decode())
        payload['timestamp'] = datetime.fromisoformat(payload['timestamp'])
        result = collection.insert_one(payload)
        print(f"[MONGO] Tersimpan {result.inserted_id} - {payload['mesin']} {payload['suhu']}°C")
    except Exception as e:
        print("Error:", e)

mqtt_client = mqtt.Client()
mqtt_client.on_connect = on_connect
mqtt_client.on_message = on_message
mqtt_client.connect("test.mosquitto.org", 1883, 60)
mqtt_client.loop_forever()

d.	Tulis skrip kedua bernama query_maintenance.py yang melakukan:
-	Mencari semua dokumen dengan biaya lebih dari 1.000.000, menampilkannya dalam bentuk DataFrame.
-	Melakukan update: pada dokumen yang memiliki mesin = "CNC-01" dan biaya = 1200000, ganti nilai teknisi menjadi "Dewi".
-	Menghitung total biaya per bulan menggunakan aggregation pipeline. Karena tanggal bertipe date, Anda perlu mengekstrak bulan dan tahun, misalnya dengan operator $dateToString atau $substr. Tampilkan hasilnya dalam bentuk tabel di konsol atau DataFrame.

import pandas as pd
from pymongo import MongoClient

# 1. Koneksi ke MongoDB
client = MongoClient('mongodb://localhost:27017/')
db = client['latihan6']
collection = db['maintenance']

print("--- LAPORAN MAINTENANCE ---")

# A. Cari biaya > 1.000.000
data_mahal = list(collection.find({"biaya": {"$gt": 1000000}}, {"_id": 0}))
df_mahal = pd.DataFrame(data_mahal)
print("\n1. Data dengan biaya > 1.000.000:")
print(df_mahal)

# B. Update teknisi CNC-01 (biaya 1.2jt) menjadi "Dewi"
collection.update_one(
    {"mesin": "CNC-01", "biaya": 1200000},
    {"$set": {"teknisi": "Dewi"}}
)
print("\n2. Update teknisi 'Dewi' berhasil dilakukan.")

# C. Agregasi Total Biaya per Bulan
pipeline = [
    {
        "$group": {
            "_id": { "$dateToString": { "format": "%Y-%m", "date": "$tanggal" } },
            "total_biaya": { "$sum": "$biaya" },
            "jumlah_tindakan": { "$sum": 1 }
        }
    },
    { "$sort": { "_id": 1 } }
]

hasil_agregasi = list(collection.aggregate(pipeline))
df_hasil = pd.DataFrame(hasil_agregasi)
if not df_hasil.empty:
    df_hasil.columns = ['Bulan', 'Total Biaya', 'Jumlah']
print("\n3. Laporan Agregasi Per Bulan:")
print(df_hasil)

client.close()

import paho.mqtt.client as mqtt
import json
from pymongo import MongoClient
from datetime import datetime

# koneksi mongoDB
client_mongo = MongoClient('mongodb://localhost:27017/')
db = client_mongo['latihan6'] # Database bebas
collection = db['produksi_mqtt']

# saat menerima pesan
def on_message(client, userdata, message):
    try:
        payload = message.payload.decode("utf-8")
        data = json.loads(payload)
        
        data['timestamp'] = datetime.now()

        reject_rate = (data['reject'] / data['jumlah']) * 100
        
        # Peringatan jika > 5%
        if reject_rate > 5:
            data['peringatan'] = True
            print(f"WARNING! Reject rate tinggi: {reject_rate:.2f}% pada Mesin {data['mesin']}")
        else:
            data['peringatan'] = False
            print(f"Data diterima: {data['mesin']} (Rate: {reject_rate:.2f}%)")

        # 5. Simpan ke MongoDB
        collection.insert_one(data)
        
    except Exception as e:
        print(f"Gagal memproses data: {e}")

broker_address = "broker.hivemq.com" # Pakai broker publik supaya pasti konek
topic = "pabrik/produksi"

client = mqtt.Client()
client.on_message = on_message

print("Menghubungkan ke broker...")
client.connect(broker_address, 1883)
client.subscribe(topic)

print(f"Subscriber aktif! Menunggu data di topik: {topic}...")
client.loop_forever() # Biar program standby terus
